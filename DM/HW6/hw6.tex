  \begin{problem}{8}
    It's true for $k=1$, let's prove that
    $\frac1{(2k+2)!} = \frac1{(2(k-1)+2)!} \frac1{2k +1} \frac1{2k + 2}$\\
    $\frac1{(2(k-1)+2)!} \frac1{2k +1} \frac1{2k + 2} =
    \frac1{(2k)!}\frac1{2k+1}\frac1{2k+2}=\frac1{(2k)!(2k+1)(2k+2)}=
    \frac1{(2k+2)!}$
  \end{problem}
