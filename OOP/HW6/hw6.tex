  \begin{problem}{3}
    \begin{minted}{eiffel}
class
  APPLICATION

inherit

  ARGUMENTS

create
  make

feature {NONE} -- Initialization
  make
    local
        i: INTEGER
        j: INTEGER
        w: INTEGER
        tail: INTEGER
        n: INTEGER
        width: INTEGER
    do
      Io.read_integer
      n := Io.last_integer

      width := (n // 2 + 1) * 4 - 1 - ((n + 1)\\ 2) * 2

      from
          i := 0
      variant
          n - i
      until
        i = n
      loop
        w := (i // 2 + 1) * 2

        from
          j := 0
          tail := width - w - 1
        variant
            width - j
        until
          j = width
        loop
          if (j < w or j > tail) and (i + j) \\ 2 = 0 then
            print ("*")
          else
              print (" ")
          end
          j := j + 1
        end

        Io.new_line
        i := i + 1
      end

    end
end
    \end{minted}
  \end{problem}
